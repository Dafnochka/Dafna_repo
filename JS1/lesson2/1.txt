1) Дан код:

var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2
при использовании префиксного инкремента он примяняется до операции присвоения.

d = b++; alert(d); // 1
постфиксный инкремент применяется к переменнной после ее записи с помощью =.

c = (2+ ++a); alert(c); // 5
здесь первый + - это операния сложения, а ++ - это префиксный инкремент, у которого высокий приоритет


d = (2+ b++); alert(d); // 4
у постфиксного инкремента приоритет выполнение ниже, чем у =,

alert(a); // 3
alert(b); // 3
